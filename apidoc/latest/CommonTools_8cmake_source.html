37 ## @brief Overloaded find_package() command.
38 #
39 # This macro calls CMake's
40 # <a href="http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:find_package">
41 # find_package()</a> command and converts obsolete all uppercase "<PKG>_<VAR>"
42 # variables to case-sensitive "<Pkg>_<VAR>" variables.
43 # It further ensures that the global variables CMAKE_FIND_LIBRARY_SUFFIXES
44 # and CMAKE_FIND_EXECUTABLE_SUFFIX are reset to the values they had before
45 # the call to find_package(). This is required if the "Find<Pkg>.cmake" module
46 # has modified these variables, but not restored their initial value.
47 macro (find_package)
48  if (BASIS_DEBUG)
49  message ("find_package(${ARGV})")
50  endif ()
51  # attention: find_package() can be recursive. Hence, use "stack" to keep
52  # track of library suffixes. Further note that we need to
53  # maintain a list of lists, which is not supported by CMake.
54  list (APPEND _BASIS_FIND_LIBRARY_SUFFIXES "{${CMAKE_FIND_LIBRARY_SUFFIXES}}")
55  list (APPEND _BASIS_FIND_EXECUTABLE_SUFFIX "${CMAKE_FIND_EXECUTABLE_SUFFIX}")
56  _find_package(${ARGV})
57  # map obsolete <PKG>_* variables to case-sensitive <Pkg>_*
58  string (TOUPPER "${ARGV0}" _FP_ARGV0_U)
59  foreach (_FP_VAR IN ITEMS FOUND DIR USE_FILE
60  VERSION VERSION_STRING VERSION_MAJOR VERSION_MINOR VERSION_PATCH
61  INCLUDE_DIR INCLUDE_DIRS INCLUDE_PATH
62  LIBRARY_DIR LIBRARY_DIRS LIBRARY_PATH)
63  if (NOT DEFINED ${ARGV0}_${_FP_VAR} AND DEFINED ${_FP_ARGV0_U}_${_FP_VAR})
64  set (${ARGV0}_${_FP_VAR} "${${_FP_ARGV0_U}_${_FP_VAR}}")
65  endif ()
66  endforeach ()
67  unset (_FP_VAR)
68  unset (_FP_ARGV0_U)
69  # restore CMAKE_FIND_LIBRARY_SUFFIXES
70  string (REGEX REPLACE ";?{([^}]*)}$" "" _BASIS_FIND_LIBRARY_SUFFIXES "${_BASIS_FIND_LIBRARY_SUFFIXES}")
71  set (CMAKE_FIND_LIBRARY_SUFFIXES "${CMAKE_MATCH_1}")
72  # restore CMAKE_FIND_EXECUTABLE_SUFFIX
73  list (LENGTH _BASIS_FIND_EXECUTABLE_SUFFIX _FP_LAST)
74  if (_FP_LAST GREATER 0)
75  math (EXPR _FP_LAST "${_FP_LAST} - 1")
76  list (REMOVE_AT _BASIS_FIND_EXECUTABLE_SUFFIX ${_FP_LAST})
77  endif ()
78  unset (_FP_LAST)
79 endmacro ()

82 ## @brief Tokenize dependency specification.
83 #
84 # This function parses a dependency specification such as
85 # "ITK-4.1{TestKernel,IO}" into the package name, i.e., ITK, the requested
86 # (minimum) package version, i.e., 4.1, and a list of package components, i.e.,
87 # TestKernel and IO. A valid dependency specification must specify the package
88 # name of the dependency (case-sensitive). The version and components
89 # specification are optional. Note that the components specification may
90 # be separated by an arbitrary number of whitespace characters including
91 # newlines. The same applies to the specification of the components themselves.
92 # This allows one to format the dependency specification as follows, for example:
93 # @code
94 # ITK {
95 # TestKernel,
96 # IO
97 # }
98 # @endcode
99 #
100 # @param [in] DEP Dependency specification, i.e., "<Pkg>[-<version>][{<Component1>[,...]}]".
101 # @param [out] PKG Package name.
102 # @param [out] VER Package version.
103 # @param [out] CMP List of package components.
104 function (basis_tokenize_dependency DEP PKG VER CMP)
105  set (CMPS)
106  if (DEP MATCHES "^([^ ]+)[ \\n\\t]*{([^}]*)}$")
107  set (DEP "${CMAKE_MATCH_1}")
108  string (REPLACE "," ";" COMPONENTS "${CMAKE_MATCH_2}")
109  foreach (C IN LISTS COMPONENTS)
110  string (STRIP "${C}" C)
111  list (APPEND CMPS ${C})
112  endforeach ()
113  endif ()
114  if (DEP MATCHES "^(.*)-([0-9]+)(\\.[0-9]+)?(\\.[0-9]+)?(\\.[0-9]+)?$")
115  set (${PKG} "${CMAKE_MATCH_1}" PARENT_SCOPE)
116  set (${VER} "${CMAKE_MATCH_2}${CMAKE_MATCH_3}${CMAKE_MATCH_4}${CMAKE_MATCH_5}" PARENT_SCOPE)
117  else ()
118  set (${PKG} "${DEP}" PARENT_SCOPE)
119  set (${VER} "" PARENT_SCOPE)
120  endif ()
121  set (${CMP} "${CMPS}" PARENT_SCOPE)
122 endfunction ()

127 # This function replaces CMake's

132 # found instead of an external package.
133 #

135 # argument was not given to this macro, a <tt>USE_&lt;Pkg&gt;</tt> option is

137 # @c OFF by the user in order to force the <tt>&lt;Pkg&gt;_FOUND</tt> variable

143 # @param [in] PACKAGE Name of other package. Optionally, the package name
144 # can include a version specification as suffix which
145 # is separated by the package name using a dash (-), i.e.,
146 # &lt;Package&gt;[-major[.minor[.patch[.tweak]]]].
147 # If a version specification is given, it is passed on as
148 # @c version argument to CMake's
149 # <a href="http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:find_package">
150 # find_package()</a> command.
151 # @param [in] ARGN Advanced arguments for
152 # <a href="http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:find_package">
153 # find_package()</a>.
154 #
155 # @retval <PACKAGE>_FOUND Whether the given package was found.

451 ## @brief Use found package.
452 #
453 # This macro includes the package's use file if the variable @c &lt;Pkg&gt;_USE_FILE
454 # is defined. Otherwise, it adds the include directories to the search path
455 # for include paths if possible. Therefore, the corresponding package
456 # configuration file has to set the proper CMake variables, i.e.,
457 # either @c &lt;Pkg&gt;_INCLUDES, @c &lt;Pkg&gt;_INCLUDE_DIRS, or @c &lt;Pkg&gt;_INCLUDE_DIR.
458 #
459 # If the given package name is the name of another module of this project
460 # (the top-level project), this function includes the use file of the specified
461 # module.
462 #
463 # @note As some packages still use all captial variables instead of ones
464 # prefixed by a string that follows the same capitalization as the
465 # package's name, this function also considers these if defined instead.
466 # Hence, if @c &lt;PKG&gt;_INCLUDES is defined, but not @c &lt;Pkg&gt;_INCLUDES, it
467 # is used in place of the latter.
468 #
469 # @note According to an email on the CMake mailing list, it is not a good idea
470 # to use basis_link_directories() any more given that the arguments to
471 # basis_target_link_libraries() are absolute paths to the library files.
472 # Therefore, this code is commented and not used. It remains here as a
473 # reminder only.
474 #
475 # @param [in] PACKAGE Name of other package. Optionally, the package name
476 # can include a version specification as suffix which
477 # is separated by the package name using a dash (-), i.e.,
478 # &lt;Package&gt;[-major[.minor[.patch[.tweak]]]].
479 # A version specification is simply ignored by this macro.
480 #
481 # @ingroup CMakeAPI
482 macro (basis_use_package PACKAGE)
483  # tokenize package specification
484  basis_tokenize_dependency ("${PACKAGE}" PKG VER CMPS)
485  # use package
486  foreach (A IN ITEMS "WORKAROUND FOR NOT BEING ABLE TO USE RETURN")
487  if (BASIS_DEBUG)
488  message ("** basis_use_package()")
489  message ("** Package: ${PKG}")
490  endif ()
491  if (PROJECT_IS_MODULE)
492  # ignore BASIS as module dependency
493  # important if BASIS itself is a project module
494  if ("^${PKG}$" STREQUAL "^BASIS$")
495  if (BASIS_DEBUG)
496  message ("** Ignoring BASIS dependency as it clearly is used already by the top-level project.")
497  endif ()
498  break ()
499  # allow modules to specify top-level project as dependency
500  elseif ("^${PKG}$" STREQUAL "^${TOPLEVEL_PROJECT_NAME}$")
501  if (BASIS_DEBUG)
502  message ("** This is the top-level project.")
503  endif ()
504  break () # instead of return()
505  # use other module of top-level project
506  else ()
507  list (FIND PROJECT_MODULES "${PKG}" PKGIDX)
508  if (NOT PKGIDX EQUAL -1)
509  if (${PKG}_FOUND)
510  if (BASIS_DEBUG)
511  message ("** Include package use file of other module.")
512  endif ()
513  include ("${${PKG}_USE_FILE}")
514  break () # instead of return()
515  else ()
516  message (FATAL_ERROR "Module ${PKG} not found! This must be an error in BASIS."
517  " Report this issue to the maintainer of this package.")
518  endif ()
519  endif ()
520  unset (PKGIDX)
521  endif ()
522  endif ()

524  # of the same superbuild as this project, set BUNDLE_PROJECT to TRUE.

533  list (FIND BUNDLE_PROJECTS "${PKG}" PKGIDX)

535  set (BUNDLE_PROJECT FALSE)

538  endif ()

615 # get_filename_component()</a> command.

622 # @returns Sets the variable named by the first argument to the requested

624 #

626 # @sa basis_get_filename_component()

3195 # ----------------------------------------------------------------------------
3196 ## @brief Process generator expressions in arguments.
3197 #
3198 # This command evaluates the $&lt;TARGET_FILE:tgt&gt; and related generator
3199 # expressions also for custom targets such as scripts and MATLAB Compiler
3200 # targets. For other generator expressions whose argument is a target name,
3201 # this function replaces the target name by the target UID, i.e., the actual
3202 # CMake target name such that the expression can be evaluated by CMake.
3203 # The following generator expressions are directly evaluated by this function:
3204 # <table border=0>
3205 # <tr>
3206 # @tp <b><tt>$&lt;TARGET_FILE:tgt&gt;</tt></b> @endtp
3207 # <td>Absolute file path of built target.</td>
3208 # </tr>
3209 # <tr>
3210 # @tp <b><tt>$&lt;TARGET_FILE_POST_INSTALL:tgt&gt;</tt></b> @endtp
3211 # <td>Absolute path of target file after installation using the
3212 # current @c CMAKE_INSTALL_PREFIX.</td>
3213 # </tr>
3214 # <tr>
3215 # @tp <b><tt>$&lt;TARGET_FILE_POST_INSTALL_RELATIVE:tgt&gt;</tt></b> @endtp
3216 # <td>Path of target file after installation relative to @c CMAKE_INSTALL_PREFIX.</td>
3217 # </tr>
3218 # </table>
3219 # Additionally, the suffix <tt>_NAME</tt> or <tt>_DIR</tt> can be appended
3220 # to the name of each of these generator expressions to get only the basename
3221 # of the target file including the extension or the corresponding directory
3222 # path, respectively.
3223 #
3224 # Generator expressions are in particular supported by basis_add_test().
3225 #
3226 # @param [out] ARGS Name of output list variable.
3227 # @param [in] ARGN List of arguments to process.
3228 #
3229 # @sa basis_add_test()
3230 # @sa http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:add_test
3231 function (basis_process_generator_expressions ARGS)
3232  set (ARGS_OUT)
3233  foreach (ARG IN LISTS ARGN)
3234  string (REGEX MATCHALL "\\$<.*TARGET.*:.*>" EXPRS "${ARG}")
3235  foreach (EXPR IN LISTS EXPRS)
3236  if (EXPR MATCHES "\\$<(.*):(.*)>")
3237  set (EXPR_NAME "${CMAKE_MATCH_1}")
3238  set (TARGET_NAME "${CMAKE_MATCH_2}")
3239  # TARGET_FILE* expression, including custom targets
3240  if (EXPR_NAME MATCHES "^TARGET_FILE(.*)")
3241  if (NOT CMAKE_MATCH_1)
3242  set (CMAKE_MATCH_1 "ABSOLUTE")
3243  endif ()
3244  string (REGEX REPLACE "^_" "" PART "${CMAKE_MATCH_1}")
3245  basis_get_target_location (ARG "${TARGET_NAME}" ${PART})
3246  # other generator expression supported by CMake
3247  # only replace target name, but do not evaluate expression
3248  else ()
3249  basis_get_target_uid (TARGET_UID "${CMAKE_MATCH_2}")
3250  string (REPLACE "${EXPR}" "$<${CMAKE_MATCH_1}:${TARGET_UID}>" ARG "${ARG}")
3251  endif ()
3252  if (BASIS_DEBUG AND BASIS_VERBOSE)
3253  message ("** basis_process_generator_expressions():")
3254  message ("** Expression: ${EXPR}")
3255  message ("** Keyword: ${EXPR_NAME}")
3256  message ("** Argument: ${TARGET_NAME}")
3257  message ("** Replaced by: ${ARG}")
3258  endif ()
3259  endif ()
3260  endforeach ()
3261  list (APPEND ARGS_OUT "${ARG}")
3262  endforeach ()
3263  set (${ARGS} "${ARGS_OUT}" PARENT_SCOPE)
3264 endfunction ()
