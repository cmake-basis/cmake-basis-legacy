48 # determine jython version string
49 if (JYTHON_EXECUTABLE)
50  execute_process (COMMAND "${JYTHON_EXECUTABLE}" -c "import sys; sys.stdout.write(';'.join([str(x) for x in sys.version_info[:3]]))"
51  OUTPUT_VARIABLE _JYTHON_VERSION
52  RESULT_VARIABLE _JYTHON_VERSION_RESULT
53  ERROR_QUIET)
54  if (_JYTHON_VERSION_RESULT EQUAL 0)
55  string (REPLACE ";" "." JYTHON_VERSION_STRING "${_JYTHON_VERSION}")
56  list (GET _JYTHON_VERSION 0 JYTHON_VERSION_MAJOR)
57  list (GET _JYTHON_VERSION 1 JYTHON_VERSION_MINOR)
58  list (GET _JYTHON_VERSION 2 JYTHON_VERSION_PATCH)
59  if (JYTHON_VERSION_PATCH EQUAL 0)
60  # it's called "Jython 2.5", not "2.5.0"
61  string (REGEX REPLACE "\\.0$" "" JYTHON_VERSION_STRING "${JYTHON_VERSION_STRING}")
62  endif()
63  else ()
64  # sys.version predates sys.version_info, so use that
65  execute_process(COMMAND "${JYTHON_EXECUTABLE}" -c "import sys; sys.stdout.write(sys.version)"
66  OUTPUT_VARIABLE _JYTHON_VERSION
67  RESULT_VARIABLE _JYTHON_VERSION_RESULT
68  ERROR_QUIET)
69  if (_JYTHON_VERSION_RESULT EQUAL 0)
70  string (REGEX REPLACE " .*" "" JYTHON_VERSION_STRING "${_JYTHON_VERSION}")
71  string (REGEX REPLACE "^([0-9]+)\\.[0-9]+.*" "\\1" JYTHON_VERSION_MAJOR "${JYTHON_VERSION_STRING}")
72  string (REGEX REPLACE "^[0-9]+\\.([0-9])+.*" "\\1" JYTHON_VERSION_MINOR "${JYTHON_VERSION_STRING}")
73  if (JYTHON_VERSION_STRING MATCHES "^[0-9]+\\.[0-9]+\\.[0-9]+.*")
74  string (REGEX REPLACE "^[0-9]+\\.[0-9]+\\.([0-9]+).*" "\\1" JYTHON_VERSION_PATCH "${JYTHON_VERSION_STRING}")
75  else ()
76  set (JYTHON_VERSION_PATCH "0")
77  endif()
78  else ()
79  set (JYTHON_VERSION_STRING)
80  set (JYTHON_VERSION_MAJOR)
81  set (JYTHON_VERSION_MINOR)
82  set (JYTHON_VERSION_PATCH)
83  endif ()
84  endif ()
85  unset (_JYTHON_VERSION)
86  unset (_JYTHON_VERSION_RESULT)
87 endif ()
