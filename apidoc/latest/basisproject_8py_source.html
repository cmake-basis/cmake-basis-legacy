270 def get_template(template, path):
271  m = re.match(r'(.*)/(\d+)\.(\d+)$', template)
272  if not m is None:
273  root = m.group(1)
274  major = int(m.group(2))
275  minor = int(m.group(3))
276  # determine available template versions and stores minor version
277  # numbers of each major version required to initialize minor number
278  # when proceeding with the previous major version
279  minors = {}
280  for f in os.listdir(root):
281  if os.path.isdir(os.path.join(root, f)):
282  m = re.match('(\d+)\.(\d+)$', f)
283  if not m is None:
284  k = int(m.group(1))
285  v = int(m.group(2))
286  if not k in minors: minors[k] = []
287  minors[k].append(v)
288  # decrease template version until template file/directory exists
289  while major > 0:
290  while minor >= 0:
291  template_path = os.path.join('%s/%d.%d' % (root, major, minor), path)
292  if os.path.exists(template_path):
293  return template_path
294  minor -= 1
295  # set minor version number to maximum minor number for next major version
296  while major > 0 and minor < 0:
297  major -= 1
298  if major in minors:
299  for next_minor in minors[major]:
300  if next_minor > minor: minor = next_minor
301  if minor < 0: break
302  # otherwise, just return input
303  return os.path.join(template, path)

599  if args.verbose > 0: sys.stderr.write(': ' + str(e))
600  sys.stderr.write('\n')
601  return -1
602  # otherwise, backup current project file using ~ suffix
603  elif args.backup:
604  try:
605  shutil.copyfile(project_path, project_path + '~')
606  except Exception, e:
607  sys.stderr.write("E %s - failed to backup file" % project_path)
608  if args.verbose > 0: sys.stderr.write(': ' + str(e))
609  sys.stderr.write('\n')
610  return -1
611  # replace project file by merged file
612  try:
613  fp = open(project_path, 'wt')
614  fp.writelines(merge['body'])
615  fp.close()
616  except Exception, e:
617  sys.stderr.write("E %s - failed to update file" % project_path)
618  if args.verbose > 0: sys.stderr.write(': ' + str(e))
619  sys.stderr.write('\n')
620  return -1
621  if retval != -1:
622  if merge['conflict']: retval = 2
623  else: retval = 1
624  # remove altered template again if update was successful
625  if not merge['conflict']:
626  try:
627  os.remove(project_path + '.template')
628  except Exception, e:
629  sys.stderr.write("E %s - failed to remove temporary file" % project_path)
630  if args.verbose > 0: sys.stderr.write(': ' + str(e))
631  sys.stderr.write('\n')
632  retval = -1
633  # cannot update files without copy of original template
634  # note that this may also be caused by a project file of
635  # same name as a newly added template file
636  else:
637  try:
638  # copy current template for reference next to the corresponding project file
639  shutil.copyfile(template_path, project_path + '.template')
640  except:
641  pass
642  sys.stderr.write("S %s - cannot add or update file: either project file of same name exists already or missing original template\n" % project_path)
643  retval = -1
644  # print status message if file was updated
645  if retval > 0:
646  if merge['conflict']: print "C %s" % project_path
647  else: print "G %s" % project_path
648  # done
649  return retval

664 def delete(path, args, isdir=False):

668  @param [in] args Namespace with command-line arguments.
669  @param [in] isdir Whether @p path is a directory path that may not have
670  a corresponding template.
671

674  @retval 1 if file/directory was removed

677  retval = 0
678

681

685  try:
686  shutil.rmtree(project_path)
687  except Exception, e:

694  os.rmdir(project_path)

699  sys.stderr.write('\n')
700  retval = -1
701  if retval != -1:
702  print "D %s" % project_path
703  retval = 1

721  return -1

732  except Exception, e:

738  if current != template:
739  if retval != -1:

748  sys.stderr.write('\n')

771 def addordel(op, path, args):
772  """Add/update or delete file or directory depending on operation specified.
773 
774  @param [in] op Operations. Either one of ADD, None, or DEL.
775  @param [in] path File path relative to template or project root, respectively.
776  @param [in] args Namespace with command-line arguments.
777 
778  @retval -1 on error
779  @retval 0 if nothing was done
780  @retval 1 if operation was successful
781 
782  """
783  # determine if path is file or directory
784  if path.endswith('/'):
785  isdir = True
786  path = path[:-1]
787  else:
788  isdir = False
789  # add file/directory to project
790  if op == ADD:
791  return add(path, args, isdir=isdir)
792  # delete file/directory from project
793  elif op == DEL:
794  return delete(path, args, isdir=isdir)
795  # nothing to do otherwise
796  return 0
