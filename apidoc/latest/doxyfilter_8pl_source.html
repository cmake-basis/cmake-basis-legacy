38 elsif ($ext eq '.py') { $lang = 'python'; }
39 elsif ($ext eq '.sh') { $lang = 'bash'; }
40 elsif ($ext =~ /\.(cmake|ctest)/) { $lang = 'cmake'; }
41 elsif ($ext eq '.m') { $lang = 'matlab'; }
42 # otherwise, consider shebang directive if given
43 if (not $lang) {
44  open FILE, $filename or die "Failed to open file \"$filename\"!";
45  $lang = $2 if <FILE> =~ /^#!\s*(\/usr\/bin\/|\/bin\/|\/usr\/bin\/env\s+)(python|perl|bash)/;
46  close FILE;
47 }
48 # create filter for source language
49 if ($lang) {
50  if ($lang eq 'bash') { $filter = new BASIS::DoxyFilter::Bash; }
51  elsif ($lang eq 'cmake') { $filter = new BASIS::DoxyFilter::CMake; }
52 }
53 # execute filter
54 if (defined $filter) {
55  $filter->process($filename);

57  $status = 0;
58 } elsif ($lang and istarget("doxyfilter-$lang")) {

61  push @cmd, $filename;
62  $status = execute(\@cmd, allow_fail=>1);
63 }
64 # otherwise, just pass input through unfiltered
