16 # @file updatefile.py
17 # @brief Update file from template file while preserving custom sections (deprecated).
18 #
19 # This script is used by the BasisUpdate.cmake module. This module is used to
20 # update files of a project instantiated from a particular revision of the
21 # BASIS project template during the configure step of CMake. This way,
22 # projects pull the changes of the compatible template automatically.
23 # Sections in the original file which are enclosed by the tokens
24 # <basis-custom> and </basis-custom> or <basis-license> and
25 # </basis-license> (without trailing spaces) are preserved while all other
26 # content is replaced by the template file. The customized sections are
27 # inserted into the template in the order they appear in the original file
28 # and the template file. If more custom sections are present in the original
29 # file than in the template file, these custom sections are appended at the
30 # end of the resulting file.
31 #
32 # See the documentation of BasisUpdate.cmake for further details.
33 #

89 def extractCustomizedSections (input):
90  custom = []
91  start = 0
92  while True:
93  start = input.find (tokenCustomStart, start)
94  if start == -1:
95  break
96  start += len (tokenCustomStart)
97  next = input.find (tokenCustomStart, start)
98  end = input.find (tokenCustomEnd, start)
99  if end == -1 or (next != -1 and end > next):
100  print "WARNING: Found begin of customized section without end token '" + tokenCustomEnd + "'"
101  print "WARNING: Will keep template section instead"
102  custom.append (tokenKeepTemplate)
103  else:
104  custom.append (input [start:end])
105  start = next
106  return custom

109 def replaceCustomizedSections (input, custom):
110  start = 0
111  end = -1
112  result = input
113  for section in custom:
114  if start != -1:

131  start = end + len (tokenCustomEnd)

173 def run (inputFile, templateFile, outputFile, force):
174  # open input and output files
175  try:
176  fIn = open (inputFile, 'r')
177  except IOError:
178  sys.stderr.write ("Failed to open file '" + inputFile + "'\n")
179  return 1
180  try:
181  fTem = open (templateFile, 'r')
182  except IOError:
183  sys.stderr.write ("Failed to open file '" + templateFile + "'\n")
184  fIn.close ()
185  return 1
186  # read files
187  input = fIn.read ()
188  fIn.close ()
189  result = fTem.read ()
190  fTem.close ()
191  # extract custom sections from input and substitute them in template
192  sections = extractCustomizedSections (input)
193  result = replaceCustomizedSections (result, sections)
194  # extract license sections from input and substitute them in template
195  sections = extractLicenseSections (input)
196  result = replaceLicenseSections (result, sections)
197  # check whether input file equals output file
198  if inputFile == outputFile:
199  # check if content changed
200  if result == input:
201  return 2
202  # query user if file should be overwritten
203  if not force:
204  try:
205  sys.stdout.write ("Template of file '" + inputFile + "' has been modified.\n")
206  sys.stdout.write ("Do you want to apply the changes (basis-custom sections remain unchanged)? ")
207  sys.stdout.flush ()
208  apply = raw_input ()
209  if apply != "y" and apply != "yes":
210  return 3
211  except EOFError:
212  return 1
213  # write result to output file if -o option given
214  if outputFile != "":
215  try:
216  fOut = open (outputFile, 'w')
217  except IOError:
218  sys.stderr.write ("Failed to open file '" + outputFile + "'\n")
